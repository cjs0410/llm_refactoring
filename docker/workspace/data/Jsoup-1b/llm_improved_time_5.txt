private void normalise(Element element) {
    List<Node> toMove = new ArrayList<Node>();
    for (Node node: element.childNodes) {
        if (node instanceof TextNode) {
            TextNode tn = (TextNode) node;
            if (!tn.isBlank())
                toMove.add(tn);
        }
    }

    for (Node node : toMove) {
        element.removeChild(node);
    }

    List<Node> existingNodes = body().childNodesCopy();
    body().empty();
    body().insertChildren(0, toMove.toArray(new Node[0]));
    body().insertChildren(toMove.size(), new TextNode(" ", ""));
    body().insertChildren(toMove.size() + 1, existingNodes.toArray(new Node[0]));
}
