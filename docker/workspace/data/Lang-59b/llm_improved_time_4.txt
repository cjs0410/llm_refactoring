public StrBuilder appendFixedWidthPadRight(Object obj, int width, char padChar) {
    if (width > 0) {
        ensureCapacity(size + width);
        String str = (obj == null ? getNullText() : obj.toString());
        int strLen = str.length();
        int endIndex = Math.min(strLen, width); // calculate the end index based on the minimum of string length and width        
        str.getChars(0, endIndex, buffer, size);
        if (strLen < width) {
            int padLen = width - strLen;
            for (int i = 0; i < padLen; i++) {
                buffer[size + endIndex + i] = padChar;
            }
        }
        size += width;
    }
    return this;
}
