private void applyPaxHeadersToCurrentEntry(Map<String, String> headers) {
    /*
     * The following headers are defined for Pax.
     * atime, ctime, charset: cannot use these without changing TarArchiveEntry fields
     * mtime
     * comment
     * gid, gname
     * linkpath
     * size
     * uid,uname
     * SCHILY.devminor, SCHILY.devmajor: don't have setters/getters for those
     */
    for (Entry<String, String> ent : headers.entrySet()) {
        String key = ent.getKey();
        String val = ent.getValue();

        switch (key) {
            case "path":
                currEntry.setName(val);
                break;
            case "linkpath":
                currEntry.setLinkName(val);
                break;
            case "gid":
                currEntry.setGroupId(parseLongWithFallback(val));
                break;
            case "gname":
                currEntry.setGroupName(val);
                break;
            case "uid":
                currEntry.setUserId(parseLongWithFallback(val));
                break;
            case "uname":
                currEntry.setUserName(val);
                break;
            case "size":
                currEntry.setSize(parseLongWithFallback(val));
                break;
            case "mtime":
                currEntry.setModTime((long) (Double.parseDouble(val) * 1000));
                break;
            case "SCHILY.devminor":
                currEntry.setDevMinor(parseIntegerWithFallback(val));
                break;
            case "SCHILY.devmajor":
                currEntry.setDevMajor(parseIntegerWithFallback(val));
                break;
        }
    }
}

private long parseLongWithFallback(String value) {
    try {
        return Long.parseLong(value);
    } catch (NumberFormatException e) {
        // Handle invalid number format
        return 0;
    }
}

private int parseIntegerWithFallback(String value) {
    try {
        return Integer.parseInt(value);
    } catch (NumberFormatException e) {
        // Handle invalid number format
        return 0;
    }
}
