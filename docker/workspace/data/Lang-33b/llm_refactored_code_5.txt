public static Class<?>[] toClass(Object[] array) {
    if (array == null) {
        return null;
    } else {
        Class<?>[] classes = new Class[array.length];
        for (int i = 0; i < array.length; i++) {
            if (array[i] == null) {
                classes[i] = null;
            } else {
                classes[i] = array[i].getClass();
            }
        }
        return classes;
    }
}
