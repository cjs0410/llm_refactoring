public Paint getPaint(double value) {
    double clampedValue = Math.max(value, this.lowerBound);
    clampedValue = Math.min(clampedValue, this.upperBound);
    int grayLevel = (int) ((clampedValue - this.lowerBound) / (this.upperBound - this.lowerBound) * 255.0);
    return new Color(grayLevel, grayLevel, grayLevel);
}
