public Paint getPaint(double value) {
    double v = value;
    if (v < this.lowerBound) {
        v = this.lowerBound;
    }
    if (v > this.upperBound) {
        v = this.upperBound;
    }
    int g = (int) ((v - this.lowerBound) / (this.upperBound - this.lowerBound) * 255.0);
    return new Color(g, g, g);
}
