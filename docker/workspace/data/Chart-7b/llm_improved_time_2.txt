private void updateBounds(TimePeriod period, int index) {

    long start = period.getStart().getTime();
    long end = period.getEnd().getTime();
    long middle = start + ((end - start) / 2);

    if (this.minStartIndex >= 0) {
        TimePeriodValue minStartItem = getDataItem(this.minStartIndex);
        long minStart = minStartItem.getPeriod().getStart().getTime();
        if (start < minStart) {
            this.minStartIndex = index;           
        }
    }
    else {
        this.minStartIndex = index;
    }

    if (this.maxStartIndex >= 0) {
        TimePeriodValue maxStartItem = getDataItem(this.maxStartIndex);
        long maxStart = maxStartItem.getPeriod().getStart().getTime();
        if (start > maxStart) {
            this.maxStartIndex = index;           
        }
    }
    else {
        this.maxStartIndex = index;
    }

    if (this.minMiddleIndex >= 0) {
        TimePeriodValue minMiddleItem = getDataItem(this.minMiddleIndex);
        TimePeriod minMiddlePeriod = minMiddleItem.getPeriod();
        long s = minMiddlePeriod.getStart().getTime();
        long e = minMiddlePeriod.getEnd().getTime();
        long minMiddle = s + (e - s) / 2;
        if (middle < minMiddle) {
            this.minMiddleIndex = index;           
        }
    }
    else {
        this.minMiddleIndex = index;
    }

    if (this.maxMiddleIndex >= 0) {
        TimePeriodValue maxMiddleItem = getDataItem(this.maxMiddleIndex);
        TimePeriod maxMiddlePeriod = maxMiddleItem.getPeriod();
        long s = maxMiddlePeriod.getStart().getTime();
        long e = maxMiddlePeriod.getEnd().getTime();
        long maxMiddle = s + (e - s) / 2;
        if (middle > maxMiddle) {
            this.maxMiddleIndex = index;           
        }
    }
    else {
        this.maxMiddleIndex = index;
    }

    if (this.minEndIndex >= 0) {
        TimePeriodValue minEndItem = getDataItem(this.minEndIndex);
        long minEnd = minEndItem.getPeriod().getEnd().getTime();
        if (end < minEnd) {
            this.minEndIndex = index;           
        }
    }
    else {
        this.minEndIndex = index;
    }

    if (this.maxEndIndex >= 0) {
        TimePeriodValue maxEndItem = getDataItem(this.maxEndIndex);
        long maxEnd = maxEndItem.getPeriod().getEnd().getTime();
        if (end > maxEnd) {
            this.maxEndIndex = index;           
        }
    }
    else {
        this.maxEndIndex = index;
    }

}
