public String get(final String name) {
    if (mapping.isEmpty()) {
        throw new IllegalArgumentException("No header mapping was specified, the record values can't be accessed by name");
    }

    final Integer index = mapping.get(name);
    if (index != null) {
        try {
            return values[index.intValue()];
        } catch (ArrayIndexOutOfBoundsException e) {
            throw new IllegalArgumentException(String.format("Index for header '%s' is %d but CSVRecord only has %d values!", name, index.intValue(), values.length));
        }
    } else {
        return null;
    }
}
