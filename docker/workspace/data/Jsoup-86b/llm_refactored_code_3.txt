public XmlDeclaration asXmlDeclaration() {
    String data = getData();
    Document doc = Jsoup.parse("<" + data.substring(1, data.length() -1) + ">", baseUri(), Parser.xmlParser());
    XmlDeclaration decl = null;
    List<Node> children = doc.children();
    if (children.size() > 0) {
        Element el = (Element) children.get(0);
        Parser parser = NodeUtils.parser(doc);
        String tagName = parser.settings().normalizeTag(el.tagName());
        boolean isXmlDeclaration = data.startsWith("!");
        decl = new XmlDeclaration(tagName, isXmlDeclaration);
        decl.attributes().addAll(el.attributes());
    }
    return decl;
}
