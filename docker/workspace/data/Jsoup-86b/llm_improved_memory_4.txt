public XmlDeclaration asXmlDeclaration() {
    String data = getData();
    String tagName = data.substring(1, data.length() - 1);
    String coreHtml = "<" + tagName + ">";
    Document doc = Jsoup.parse(coreHtml, baseUri(), Parser.xmlParser());
    XmlDeclaration decl = null;
    List<Element> children = doc.children();
    if (!children.isEmpty()) {
        Element el = children.get(0);
        String normalizedTag = NodeUtils.parser(doc).settings().normalizeTag(el.tagName());
        boolean isXmlDeclaration = data.startsWith("!");
        decl = new XmlDeclaration(normalizedTag, isXmlDeclaration);
        decl.attributes().addAll(el.attributes());
    }
    return decl;
}
