public static boolean containsAny(CharSequence cs, char[] searchChars) {
    if (isEmpty(cs) || ArrayUtils.isEmpty(searchChars)) {
        return false;
    }
    int csLength = cs.length();
    int searchLength = searchChars.length;
    int csIndex = 0;
    int searchIndex = 0;

    while (csIndex < csLength) {
        char ch = cs.charAt(csIndex);
        if (searchChars[searchIndex] == ch) {
            if (searchIndex == searchLength - 1) {
                if (Character.isHighSurrogate(ch) && csIndex < csLength - 1) {
                    char ch2 = cs.charAt(csIndex + 1);
                    if (Character.isLowSurrogate(ch2)) {
                        return true;
                    }
                } else {
                    return true;
                }
            }
            searchIndex++;
        } else {
            searchIndex = 0;
            if (searchChars[0] == ch) {
                searchIndex = 1;
            }
        }
        csIndex++;
    }

    return false;
}
