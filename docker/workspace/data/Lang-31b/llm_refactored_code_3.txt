public static boolean containsAny(CharSequence cs, char[] searchChars) {
    if (isEmpty(cs) || ArrayUtils.isEmpty(searchChars)) {
        return false;
    }
    int csLength = cs.length();
    int searchLength = searchChars.length;
    int csLastIndex = csLength - 1;
    int searchLastIndex = searchLength - 1;

    for (int i = 0; i < csLength; i++) {
        char ch = cs.charAt(i);

        if (Character.isHighSurrogate(ch) && i < csLastIndex && searchLastIndex > 0) {
            char nextCh = cs.charAt(i + 1);
            char nextSearchCh = searchChars[1];

            if (nextCh == nextSearchCh) {
                return true;
            }
        } else {
            for (int j = 0; j < searchLength; j++) {
                if (searchChars[j] == ch) {
                    return true;
                }
            }
        }
    }
    return false;
}
