public static boolean containsAny(CharSequence cs, char[] searchChars) {
    if (isEmpty(cs) || ArrayUtils.isEmpty(searchChars)) {
        return false;
    }

    int csLength = cs.length();
    int searchLength = searchChars.length;
    int csLastIndex = csLength - 1;
    int searchLastIndex = searchLength - 1;

    for (int i = 0; i < csLength; i++) {
        char ch = cs.charAt(i);

        for (int j = 0; j < searchLength; j++) {
            if (searchChars[j] == ch) {
                boolean isSupplementaryChar = (i < csLastIndex && j < searchLastIndex && ch >= Character.MIN_HIGH_SURROGATE && ch <= Character.MAX_HIGH_SURROGATE);

                if (isSupplementaryChar && searchChars[j + 1] == cs.charAt(i + 1)) {
                    return true;
                } else if (!isSupplementaryChar) {
                    return true;
                }
            }
        }
    }

    return false;
}
