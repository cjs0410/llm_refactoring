private void writeBits(final DataOutput header, final BitSet bits, final int length) throws IOException {
    int shift = 7;
    byte cache = 0;
    for (int i = 0; i < length; i++) {
        cache |= ((bits.get(i) ? 1 : 0) << shift);
        if (--shift < 0) {
            header.writeByte(cache);
            shift = 7;
            cache = 0;
        }
    }
    if (shift != 7) {
        header.writeByte(cache);
    }
}
