private void writeBits(final DataOutput header, final BitSet bits, final int length) throws IOException {
    int shift = 7;
    long mask = 1L << shift;
    int cache = 0;

    for (int i = 0; i < length; i++) {
        cache |= (bits.get(i) ? mask : 0);

        if ((mask >>= 1) == 0) {
            header.write(cache);
            shift = 7;
            cache = 0;
            mask = 1L << shift;
        }
    }

    if (shift != 7) {
        header.write(cache);
    }
}
