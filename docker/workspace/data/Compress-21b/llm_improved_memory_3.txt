private void writeBits(final DataOutput header, final BitSet bits, final int length) throws IOException {
    int byteCache = 0;
    int shift = 7;
    int bytePosition = 0;

    for (int i = 0; i < length; i++) {
        byteCache |= (bits.get(i) ? 1 : 0) << shift;
        shift--;

        if (shift < 0) {
            header.write(byteCache);

            byteCache = 0;
            shift = 7;
            bytePosition++;
        }
    }
    
    if (bytePosition < length) {
        header.write(byteCache);
    }
}
