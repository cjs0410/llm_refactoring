public String absUrl(String attributeKey) {
    Validate.notEmpty(attributeKey);

    String relUrl = attr(attributeKey);
    if (!hasAttr(attributeKey)) {
        return ""; // nothing to make absolute with
    } else {
        try {
        URL base = new URL(baseUri);

        if (relUrl.startsWith("?")) {
            StringBuilder absUrl = new StringBuilder(base.toString());
            // Remove any query parameters from the base URL
            int queryIndex = absUrl.indexOf("?");
            if (queryIndex != -1) {
                absUrl = absUrl.delete(queryIndex, absUrl.length());
            }
            // Append the relative URL starting with '?'
            absUrl.append(relUrl);
            return absUrl.toString();
        }

        URL abs = new URL(base, relUrl);
        return abs.toExternalForm();

        } catch (MalformedURLException e) {
            return "";
        }
    }
}
