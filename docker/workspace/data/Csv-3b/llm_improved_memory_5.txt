int readEscape() throws IOException {
    // the escape char has just been read (normally a backslash)
    final int c = in.read();
    switch (c) {
        case 'r':
            return CharUtils.CR;
        case 'n':
            return CharUtils.LF;
        case 't':
            return CharUtils.TAB;
        case 'b':
            return CharUtils.BACKSPACE;
        case 'f':
            return CharUtils.FF;
        case CharUtils.CR:
        case CharUtils.LF:
        case CharUtils.FF:
        case CharUtils.TAB:
        case CharUtils.BACKSPACE:
            return c;
        case END_OF_STREAM:
            throw new IOException("EOF whilst processing escape sequence");
        default:
            // Now check for meta-characters
            if (isDelimiter(c) || isEscape(c) || isQuoteChar(c) || isCommentStart(c)) {
                return c;
            }
            // Indicate unexpected char
            return END_OF_STREAM;
    }
}
