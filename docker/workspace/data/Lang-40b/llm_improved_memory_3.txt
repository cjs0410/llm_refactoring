public static boolean containsIgnoreCase(String str, String searchStr) {
    if (str == null || searchStr == null) {
        return false;
    }
    int len = searchStr.length();
    int max = str.length() - len;
    for (int i = 0; i <= max; i++) {
        if (compareIgnoreCase(str, i, searchStr, len)) {
            return true;
        }
    }
    return false;
}

private static boolean compareIgnoreCase(String str, int offset, String searchStr, int len) {
    for (int i = 0; i < len; i++) {
        char ch1 = Character.toLowerCase(str.charAt(offset + i));
        char ch2 = Character.toLowerCase(searchStr.charAt(i));
        if (ch1 != ch2) {
            return false;
        }
    }
    return true;
}
