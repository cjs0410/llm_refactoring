protected void setName(String name) {
    if (name != null && getPlatform() == PLATFORM_FAT && !name.contains("/")) {
        char[] chars = name.toCharArray();
        for (int i = 0; i < chars.length; i++) {
            if (chars[i] == '\\') {
                chars[i] = '/';
            }
        }
        name = new String(chars);
    }
    this.name = name;
}
