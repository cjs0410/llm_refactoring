protected void setName(String name) {
    if (name != null && getPlatform() == PLATFORM_FAT && name.indexOf("/") == -1) {
        StringBuilder sb = new StringBuilder(name);
        for (int i = 0; i < sb.length(); i++) {
            if (sb.charAt(i) == '\\') {
                sb.setCharAt(i, '/');
            }
        }
        this.name = sb.toString();
    } else {
        this.name = name;
    }
}
