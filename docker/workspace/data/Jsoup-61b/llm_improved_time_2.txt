public boolean hasClass(String className) {
    final String classAttr = attributes.getIgnoreCase("class");
    final int len = classAttr != null ? classAttr.length() : 0;
    final int wantLen = className.length();

    if (len == 0 || len < wantLen) {
        return false;
    }

    // if both lengths are equal, only need to compare the className with the attribute
    if (len == wantLen) {
        return className.equalsIgnoreCase(classAttr);
    }

    // otherwise, scan for whitespace and compare regions (with no string or ArrayList allocations)
    int start = 0;
    for (int i = 0; i < len; i++) {
        if (Character.isWhitespace(classAttr.charAt(i))) {
            if (i - start == wantLen && classAttr.regionMatches(true, start, className, 0, wantLen)) {
                return true;
            }
            start = i + 1;
        }
    }

    // check the last entry
    if (len - start == wantLen && classAttr.regionMatches(true, start, className, 0, wantLen)) {
        return true;
    }

    return false;
}
