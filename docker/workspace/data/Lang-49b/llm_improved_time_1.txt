    public Fraction reduce() {
        if (numerator == 0) {
            if (equals(ZERO)) {
                return this;
            } else {
                return ZERO;
            }
        }
        int gcd = greatestCommonDivisor(Math.abs(numerator), denominator);
        if (gcd == 1) {
            return this;
        }
        return Fraction.getFraction(numerator / gcd, denominator / gcd);
    }
