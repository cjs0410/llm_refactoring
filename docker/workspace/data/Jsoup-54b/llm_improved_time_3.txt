private void copyAttributes(org.jsoup.nodes.Node source, Element el) {
    for (Attribute attribute : source.attributes()) {
        String key = sanitizeAttributeKey(attribute.getKey());
        if (isValidAttributeKey(key)) {
            el.setAttribute(key, attribute.getValue());
        }
    }
}

private static String sanitizeAttributeKey(String key) {
    return key.replaceAll("[^-a-zA-Z0-9_:.]", "");
}

private static boolean isValidAttributeKey(String key) {
    return key.matches("[a-zA-Z_:]{1}[-a-zA-Z0-9_:.]*");
}
