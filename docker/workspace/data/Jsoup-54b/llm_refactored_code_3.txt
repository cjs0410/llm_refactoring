private void copyAttributes(org.jsoup.nodes.Node source, Element el) {
    for (Attribute attribute : source.attributes()) {
        String key = cleanAttributeKey(attribute.getKey());
        if (isValidXmlAttributeName(key)) {
            el.setAttribute(key, attribute.getValue());
        }
    }
}

private String cleanAttributeKey(String attributeName) {
    // Remove invalid characters from attribute name
    return attributeName.replaceAll("[^-a-zA-Z0-9_:.]", "");
}

private boolean isValidXmlAttributeName(String attributeName) {
    // Valid xml attribute names are: ^[a-zA-Z_:][-a-zA-Z0-9_:.]
    String regex = "[a-zA-Z_:]{1}[-a-zA-Z0-9_:]*";
    return attributeName.matches(regex);
}
