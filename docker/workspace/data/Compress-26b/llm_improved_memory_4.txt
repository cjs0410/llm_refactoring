    public static long skip(InputStream input, long numToSkip) throws IOException {
        long available = numToSkip;

        while (numToSkip > 0) {
            long skipped = input.skip(numToSkip);
            if (skipped == 0) {
                break;
            }
            numToSkip -= skipped;
        }
            
        if (numToSkip > 0) {
            int bufferSize = (int) Math.min(numToSkip, SKIP_BUF_SIZE);
            byte[] skipBuf = new byte[bufferSize];
            int bytesRead = 0;

            while (numToSkip > 0 && (bytesRead = readFully(input, skipBuf, 0, bufferSize)) > 0) {
                numToSkip -= bytesRead;
            }
        }
        return available - numToSkip;
    }
