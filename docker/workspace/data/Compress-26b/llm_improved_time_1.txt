public static long skip(InputStream input, long numToSkip) throws IOException {
    long available = numToSkip;

    while (numToSkip > 0) {
        long skipped = input.skip(numToSkip);
        if (skipped == 0) {
            break;
        }
        numToSkip -= skipped;
    }

    byte[] skipBuf = new byte[SKIP_BUF_SIZE];
    while (numToSkip > 0) {
        int readLen = (int) Math.min(numToSkip, SKIP_BUF_SIZE);
        int skippedLen = input.readNBytes(skipBuf, 0, readLen);
        if (skippedLen < 1) {
            break;
        }
        numToSkip -= skippedLen;
    }

    return available - numToSkip;
}
