public static long skip(InputStream input, long numToSkip) throws IOException {
    long available = numToSkip;
    while (numToSkip > 0) {
        long skipped = input.skip(numToSkip);
        if (skipped == 0) {
            break;
        }
        numToSkip -= skipped;
    }

    if (numToSkip > 0) {
        byte[] skipBuf = new byte[SKIP_BUF_SIZE];
        int numSkipped;
        while (numToSkip > 0) {
            numSkipped = input.read(skipBuf, 0, (int) Math.min(numToSkip, SKIP_BUF_SIZE));
            if (numSkipped < 0) {
                break;
            }
            numToSkip -= numSkipped;
        }
    }
    return available - numToSkip;
}
